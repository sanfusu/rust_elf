use crate::elf::{Elf32, Elf64, ElfBasicType};

/// 各字段含义等同于 [`Sym32`]，唯一不同的是各字段大小和内存布局。
#[repr(packed)]
pub struct Sym64 {
    pub name: <Elf64 as ElfBasicType>::Word,
    pub info: u8,
    pub other: u8,
    pub shndx: <Elf64 as ElfBasicType>::Half,
    pub value: <Elf64 as ElfBasicType>::Addr,
    pub size: <Elf64 as ElfBasicType>::Xword,
}

/// 符号表条目
#[repr(packed)]
pub struct Sym32 {
    /// 用来保存对象文件中符号字符串表中的索引
    pub name: <Elf32 as ElfBasicType>::Word,
    /// 该成员用来保存与符号相关联的值。
    /// 既可以是绝对值，也可以是一个地址，具体是哪一个取决于上下文环境。
    pub value: <Elf32 as ElfBasicType>::Addr,
    /// 符号相关联的大小。如果是 0，表示没有大小，或者大小未知。
    pub size: <Elf32 as ElfBasicType>::Word,
    /// 用来表示符号类型和绑定的属性，可用值见 [`StBind`] 和 [`StType`]
    pub info: u8,
    /// 当前值为 0，属于保留字段
    pub other: u8,
    /// 每一个符号表条目都和一些 section 相关。
    /// 该字段保存相关联的 section 在 section header table 中的索引
    pub shndx: <Elf32 as ElfBasicType>::Half,
}

#[derive(Ordinalize, Debug)]
#[repr(u8)]
pub enum StBindNormal {
    LOCAL = 0,
    GLOABL,
    WEAK,
    LOPROC = 13,
    HIPROC = 15,
}
pub enum StBind {
    Bind(StBindNormal),
    ProcessorSpecific(u8),
}
#[derive(Ordinalize, Debug)]
#[repr(u8)]
pub enum StTypeNormal {
    NOTYPE = 0,
    OBJECT = 1,
    FUNC = 2,
    SECTION = 3,
    FILE = 4,
    LOPROC = 13,
    HIPROC = 15,
}

pub enum StType {
    Type(StTypeNormal),
    ProcessorSpecific(u8),
}

pub enum Sym {
    Symbol32(Sym32),
    Symbol64(Sym64),
}
