pub mod sp_idx {
    //! 特殊的 section 索引

    /// 用于标记一个未定义或无意义的 section
    pub const UNDEF: usize = 0;
    pub const LOPROC: usize = 0xff00;
    pub const HIPROC: usize = 0xff1f;
    pub const LOOS: usize = 0xff3f;
    pub const HIOS: usize = 0xfff1;
    /// 表示相应的引用是绝对值
    pub const ABS: usize = 0xfff1;
    /// 表示符号被声明为 common block。（Fortan COMMON 或者 C 中的临时声明）
    pub const COMMON: usize = 0xfff2;
}

pub mod sh_attrs {
    use core::convert::TryFrom;

    use super::Xword;
    #[derive(Default, MetaData)]
    pub struct ShFlag {
        value: Xword,
    }
    /// Section 包含可写数据
    pub const WRITE: ShFlag = ShFlag::new(0x1);
    /// Section 会在程序内存映像中分配空间
    pub const ALLOC: ShFlag = ShFlag::new(0x2);
    /// Section 包含可执行指令
    pub const EXECINSTR: ShFlag = ShFlag::new(0x4);

    impl ShFlag {
        const fn new(value: Xword) -> Self {
            Self { value }
        }
    }

    impl Into<Xword> for ShFlag {
        fn into(self) -> Xword {
            self.value
        }
    }
    impl TryFrom<Xword> for ShFlag {
        type Error = &'static str;

        fn try_from(value: Xword) -> Result<Self, Self::Error> {
            if value == 0x1
                || value == 0x2
                || value == 0x4
                || (value & 0x0f00_0000 != 0)
                || (value & 0xf000_0000 != 0)
            {
                Ok(ShFlag::new(value))
            } else {
                Err("Out of range")
            }
        }
    }
}
#[derive(MetaData, Layout)]
#[repr(packed)]
pub struct Shdr {
    /// Section 名称，为 section 名称字符串表中的偏移地址
    pub sh_name: Word,
    /// Section 类型
    pub sh_type: ShType,
    /// Section 属性
    pub sh_flags: ShFlag,
    /// Section 在内存中的起始地址（虚拟地址）。
    /// 如果 section 没有在程序映像中分配空间，则该字段为 0。
    pub sh_addr: Addr,
    /// Section 起始内容在文件中偏移地址。
    pub sh_offset: Off,
    /// Section 的大小。
    pub sh_size: Xword,
    /// 包含关联 section 的索引。关联的 section 和本 section 的类型有关：
    ///
    /// | Section 类型 | 关联的 section |
    /// |:- |:-|
    /// | [`DYNAMIC`](sh_type::DYNAMIC) | 本 section 所使用的字符串表 |
    /// | [`HASH`](sh_type::HASH) | hash table 所适用的符号表  |
    /// | [`REL`](sh_type::REL) | 重定位所引用的符号表 |
    /// | [`RELA`](sh_type::RELA) | |
    /// | [`SYMTAB`](sh_type::SYMTAB) | 本 section 中的元素所使用的字符串表 |
    /// | [`DYNSYM`](sh_type::DYNSYM) | |
    /// | 其他 | [`UNDEF`](sp_idx::UNDEF) |
    pub sh_link: Word,
    /// 包含关于 section 的额外信息
    pub sh_info: Word,
    /// 包含 section 的对齐要求。
    pub sh_addralign: Xword,
    /// 如果 section 中包含固定大小的元素，则该字段为元素的大小，否则该字段值为 0
    pub sh_entsize: Xword,
}
