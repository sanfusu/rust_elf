use super::{Addr, Half, Off, Word};

// E_IDENT field type
define_transparent_meta_data! {
    pub (ObjectFileClass, u8) {
        Wellknown: {
            pub ELF_CLASS32: 1
            pub ELF_CLASS64: 2
        }
        ValidRange: {
            ELF_CLASS_RANGE: ELF_CLASS32 ..= ELF_CLASS64
        }
        Default: ELF_CLASS64
    }
    pub (DataEncoding, u8) {
        Wellknown: {
            pub ELF_LSB: 1
            pub ELF_MSB: 2
        }
        ValidRange:{
            ELF_DATA_ENCODING_RANGE: ELF_LSB..=ELF_MSB
        }
        Default: ELF_LSB
    }
    pub (IdentVersion, u8) {
        Wellknown: {
            pub EIDV_CURRENT: 1
        }
        ValidRange: {
            ELF_IDENT_VERSION_RANGE: EIDV_CURRENT..=EIDV_CURRENT
        }
        Default: EIDV_CURRENT
    }
    pub (OsAbiIdentifer, u8) {
        Wellknown: {
            pub ELFOSABI_SYS: 0
            pub ELFOSABI_HPUX: 1
            pub ELFOSABI_STANDALONE: 255
        }
        ValidRange: {
            ELFOSABI_HOST: ELFOSABI_SYS..=ELFOSABI_HPUX
            ELFOSABI_STANDLONE: ELFOSABI_STANDALONE..=ELFOSABI_STANDALONE
        }
        Default: ELFOSABI_SYS
    }
}

define_transparent_meta_data! {
    pub (ObjectFileType, Half) {
        Wellknown: {
            pub ET_NONE: 0
            pub ET_REL: 1
            pub ET_EXEC: 2
            pub ET_DYN: 3
            pub ET_CORE: 4
            ET_LOOS: 0xfe00
            ET_HIOS: 0xfeff
            ET_LOPROC: 0xff00
            ET_HIPROC: 0xffff
        }
        ValidRange: {
            ET_RESERVED_RANGE: ET_NONE..=ET_CORE
            ET_EV_SPECIFIC_RANGE: ET_LOOS..=ET_HIOS
            ET_PROCESSOR_SPECIFIC_RANGE: ET_LOPROC..=ET_HIPROC
        }
        Default: ET_NONE
    }
    pub (FileVersion, Word) {
        Wellknown: {
            pub EV_CURRENT: 1
        }
        ValidRange: {
            ELF_FILE_VERSION_RANGE: EV_CURRENT..=EV_CURRENT
        }
        Default: EV_CURRENT
    }
}
#[derive(MetaData, Default)]
pub struct Ident {
    pub mag: [u8; 4],
    pub file_class: ObjectFileClass,
    pub data_encode: DataEncoding,
    /// 用来标识对象文件格式，目前只能是 EIDV_CURRENT，等同于 [`Ehdr::e_version`]
    pub ident_version: IdentVersion,
    /// 用来标识操作系统和相应的 abi。
    /// 部分字段的解释以来于该字段。
    pub osabi: OsAbiIdentifer,
    /// 该字段用于区分同一个 abi 下的不同版本， 版本号的解释依赖于  [`Ident::osabi`]
    pub abi_version: u8,
}
#[derive(MetaData, Ehdr)]
#[repr(packed)]
pub struct Ehdr {
    pub e_ident: Ident,
    _pad: [u8; 16 - core::mem::size_of::<Ident>()],
    /// 标识对象文件类型
    pub e_type: ObjectFileType,
    /// 标识目标架构
    pub e_machine: Half,
    /// 标识对象文件格式，等同于 [`Ident::ident_version`]
    pub e_version: FileVersion,
    /// 包含程序入口的虚拟地址。如果没有入口点，该字段为 0.
    pub e_entry: Addr,
    #[phoff]
    /// 包含 program header table 的文件偏移
    pub e_phoff: Off,
    #[shoff]
    /// 包含 section header table 的文件偏移
    pub e_shoff: Off,
    /// 包含处理器相关的 flag
    pub e_flags: Word,
    /// 包含 ELF header 的大小
    pub e_ehsize: Half,
    #[phentsize]
    /// 包含 program header table 中每个元素的大小
    pub e_phentsize: Half,
    #[phnum]
    /// program header table 中元素的数量
    pub e_phnum: Half,
    #[shentsize]
    /// section header table 中每个元素的大小
    pub e_shentsize: Half,
    #[shnum]
    /// section header table 中元素的数量
    pub e_shnum: Half,
    /// section 名称字符串表所在的 section 在 section header table 中的索引。
    /// 如果没有 section 名称字符串表，该字段值为 [`SHN_UNDEF`](super::section::sp_idx::UNDEF)
    pub e_shstrndx: Half,
}
